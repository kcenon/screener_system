# 가치평가
- ev_sales                        # EV/Sales
- ev_fcf                          # EV/FCF
- ev_ebit                         # EV/EBIT
- price_to_book_tangible          # P/TBV
- earnings_yield                  # Earnings Yield
- fcf_yield                       # FCF Yield

# 퀄리티
- roic                            # Return on Invested Capital
- roce                            # Return on Capital Employed
- gross_profit_to_assets          # Gross Profitability
- asset_turnover                  # 자산회전율
- inventory_turnover              # 재고자산회전율
- receivables_turnover            # 매출채권회전율

# 안정성/리스크
- altman_z_score                  # 파산 위험 지표
- beneish_m_score                 # 분식회계 탐지
- piotroski_f_score               # Piotroski F-Score
- interest_coverage_ratio         # 이자보상배율
- cash_ratio                      # 현금비율
- quick_ratio                     # 당좌비율

# 성장성
- revenue_cagr_3y                 # 3년 매출 CAGR
- revenue_cagr_5y                 # 5년 매출 CAGR
- eps_cagr_3y                     # 3년 EPS CAGR
- eps_cagr_5y                     # 5년 EPS CAGR
- dividend_cagr_5y                # 5년 배당 CAGR

# 배당
- dividend_payout_ratio           # 배당성향
- dividend_growth_rate            # 배당성장률
